-- employees table query

-- a. Write a query to select all columns and rows from the employees table.
Description: Selects all rows and columns from employees table.

    SELECT * FROM employees;


-- b. Write a query to select only the name and salary columns of all employees with a salary greater than 50000.
Description: finds all employees with name and salary from employees table whose salary greater than 50000

    SELECT name, salary FROM employees WHERE salary > 50000;


-- c. Write a query to calculate the average salary of all employees.
Description: finds average salary from employees table by usign avg() aggregate function on salary column 

    SELECT AVG(salary) AS avg_salary FROM employees;


-- d. Write a query to count the number of employees who work in the "Marketing" department.
Description: Joins two tables with department_id in employees table and id in departments table and counts the number of rows where department name is equal "Marketing".

    SELECT COUNT(*) AS total_employees, departments.name AS department_name FROM employees JOIN departments ON employees.department_id= departments.id WHERE departments.name = "Marketing";


-- e. Write a query to update the salary column of the employee with an id of 1001 to 60000.
Description: Update the salary of employee whose id is equal `1001`. 

    UPDATE employees SET salary = 60000 WHERE id=1001;


-- f. Write a query to delete all employees whose salary is less than 30000.
Description: Delete all employees from employees table whose salary is less than 30000

    DELETE FROM employees WHERE salary < 30000;



-- departments table query

-- a. Write a query to select all columns and rows from the departments table.
Description: Finds all departments from  departments table;

    SELECT * FROM departments;


-- b. Write a query to select only the name and manager columns of the "Finance" department.
Description: Finds name and manager from departments table where name="Finance";

    SELECT name, manager FROM departments WHERE name="Finance";


-- c. Write a query to calculate the total number of employees in each department.
Description: Finds total employees in each department by right joining employees table with departments table so that we can get all departments names which does not have any employees too.

    SELECT COUNT(employees.id) AS total_employees, departments.name AS department_name FROM employees RIGHT JOIN departments ON employees.department_id = departments.id GROUP BY departments.id;


-- d. Write a query to insert a new department called "Research" with a manager named "John Doe".
Description: Insert a new row in departments table with name="Research" and manager="John Doe" columns.

    INSERT INTO departments (name, manager) VALUES ("Research", "John Doe");
